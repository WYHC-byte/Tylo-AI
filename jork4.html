Here's a JavaScript function that generates random passwords with customizable options:

/**
 * Generates a random password with customizable options
 * @param {number} length - The length of the password (default: 12)
 * @param {Object} options - Configuration options
 * @param {boolean} options.lowercase - Include lowercase letters (default: true)
 * @param {boolean} options.uppercase - Include uppercase letters (default: true)
 * @param {boolean} options.numbers - Include numbers (default: true)
 * @param {boolean} options.symbols - Include special symbols (default: true)
 * @returns {string} The generated password
 */
function generatePassword(length = 12, options = {}) {
  // Set default options
  const config = {
    lowercase: options.lowercase !== false,
    uppercase: options.uppercase !== false,
    numbers: options.numbers !== false,
    symbols: options.symbols !== false
  };

  // Define character sets
  const charSets = {
    lowercase: 'abcdefghijklmnopqrstuvwxyz',
    uppercase: 'ABCDEFGHIJKLMNOPQRSTUVWXYZ',
    numbers: '0123456789',
    symbols: '!@#$%^&*()_+-=[]{}|;:,.<>?'
  };

  // Create character pool based on selected options
  let charPool = '';
  let requiredChars = [];
  
  Object.keys(config).forEach(key => {
    if (config[key]) {
      charPool += charSets[key];
      // Store one character from each selected set for guaranteed inclusion
      requiredChars.push(charSets[key][Math.floor(Math.random() * charSets[key].length)]);
    }
  });

  // Ensure we have at least one character type selected
  if (charPool === '') {
    throw new Error('At least one character type must be selected');
  }

  // Generate the password
  let password = '';
  
  // First add required characters (at least one from each selected type)
  for (let i = 0; i < requiredChars.length; i++) {
    password += requiredChars[i];
  }
  
  // Then fill the rest with random characters from the pool
  for (let i = password.length; i < length; i++) {
    const randomIndex = Math.floor(Math.random() * charPool.length);
    password += charPool[randomIndex];
  }

  // Shuffle the password to avoid predictable positions of required chars
  return shuffleString(password);
}

// Helper function to shuffle a string
function shuffleString(string) {
  const array = string.split('');
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
  return array.join('');
}

// Example usage:
console.log(generatePassword()); // Default 12-char password with all options
console.log(generatePassword(16, { symbols: false })); // 16-char password without symbols
console.log(generatePassword(8, { lowercase: true, uppercase: true, numbers: false, symbols: false })); // 8-char letters-only password

This function ensures that each password includes at least one character from each selected character type, making it more secure. The shuffling step ensures that required characters aren't always in predictable positions.
